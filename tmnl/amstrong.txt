echo "Enter the number "
read num

temp=$num

sum=0

count=${#num}

while [ $temp -ne 0 ]
do
        digit=$((temp%10))
        pow=1
        for((i=0;i<count;i++))
        do
                pow=$((pow*digit))
        done

        sum=$((sum+pow))
        temp=$((temp/10))
done

if [ $sum -eq $num ]
then
        echo "It is Amstrong number"
else
        echo "it is not an Amstrong number"
fi
