#include <stdio.h>
#include <signal.h>
#include <unistd.h>
#include <stdlib.h>

void handler(int sig) {
    printf("Child received signal %d from parent\n", sig);
    exit(0); 
}

int main() {
    signal(SIGUSR1, handler); 
    pid_t pid;

    pid = fork();

    if (pid < 0) {
        printf("Fork failed\n");
        return 1;
    }

    if (pid == 0) {
        // Child process
        printf("Child is waiting for signal...\n");
        while (1) {
            pause();   
        }
    } else {
        // Parent process
        printf("Parent will send signal to child after 3 seconds...\n");
        sleep(3);                        
        kill(pid, SIGUSR1);             
        printf("Parent sent signal to child\n");
    }

    return 0;
}
